class Cl01(val nombre: String,
  val color:String) {
 fun saluda() {
  println(
   "Hola. Soy $nombre - $color")
 }
}
fun main() {
 val tel1 = Cl01("pp", "marrón")
 val tel2 = Cl01("qk", "verde")
 tel1.saluda()
 println(tel1.nombre)
 println(tel2.nombre)
 println(tel2.color)
 println(tel1.color)
 tel2.saluda()
}

B. Setters y getters

Revísalo en Kotlin play.

class ClX {
 var nombre = ""
  get() = field
  set (nombre) {
   field = nombre
  }
  val juega
   get() = "${this.nombre} juega"
}

fun main() {
 val tt1 = ClX()
 tt1.nombre = "pp"
 println(tt1.nombre)
 println(tt1.juega)
}
